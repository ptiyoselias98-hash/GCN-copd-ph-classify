### Step 1.2 — Modify the model: HybridGCN

```python
class HybridGCN(nn.Module):
    """GCN graph embedding + commercial CT radiomics → MLP → classification."""
    
    def __init__(self, gcn_in=12, gcn_hidden=64, radiomics_dim=45, out=2, num_layers=3, dropout=0.3):
        super().__init__()
        # GCN backbone (same as before)
        self.convs = nn.ModuleList()
        self.bns = nn.ModuleList()
        self.convs.append(SAGEConv(gcn_in, gcn_hidden))
        self.bns.append(BatchNorm(gcn_hidden))
        for _ in range(num_layers - 1):
            self.convs.append(SAGEConv(gcn_hidden, gcn_hidden))
            self.bns.append(BatchNorm(gcn_hidden))
        
        self.pool = global_mean_pool
        
        # Radiomics branch
        self.radiomics_bn = nn.BatchNorm1d(radiomics_dim)
        
        # Fusion MLP: gcn_hidden + radiomics_dim → classification
        fused_dim = gcn_hidden + radiomics_dim  # 64 + 45 = 109
        self.classifier = nn.Sequential(
            nn.Linear(fused_dim, 64),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(64, 32),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(32, out)
        )
    
    def forward(self, x, edge_index, batch, radiomics):
        # GCN forward
        for conv, bn in zip(self.convs, self.bns):
            x = F.relu(bn(conv(x, edge_index)))
            x = F.dropout(x, p=0.3, training=self.training)
        
        graph_emb = self.pool(x, batch)          # (B, 64)
        rad_emb = self.radiomics_bn(radiomics)   # (B, 45)
        
        # Fusion
        fused = torch.cat([graph_emb, rad_emb], dim=1)  # (B, 109)
        return self.classifier(fused)
```

### Implementation in skeleton.py:

```python
def compute_curvature(path, spacing):
    """Compute mean curvature along a branch path."""
    if len(path) < 3:
        return 0.0
    # Tangent vectors
    phys_path = path.astype(float) * np.array(spacing)
    tangents = np.gradient(phys_path, axis=0)
    tangent_norms = np.linalg.norm(tangents, axis=1, keepdims=True)
    tangent_norms[tangent_norms < 1e-8] = 1e-8
    unit_tangents = tangents / tangent_norms
    # Curvature = rate of change of unit tangent
    curvature_vectors = np.gradient(unit_tangents, axis=0)
    curvatures = np.linalg.norm(curvature_vectors, axis=1)
    return float(np.mean(curvatures))
```

### Lobe assignment:

```python
def assign_lobe(centroid_voxel, lobe_mask):
    """Map a 3D coordinate to its lung lobe."""
    z, y, x = int(centroid_voxel[0]), int(centroid_voxel[1]), int(centroid_voxel[2])
    if 0 <= z < lobe_mask.shape[0] and 0 <= y < lobe_mask.shape[1] and 0 <= x < lobe_mask.shape[2]:
        label = lobe_mask[z, y, x]
        # Standard lobe labels: 1=RUL, 2=RML, 3=RLL, 4=LUL, 5=LLL
        one_hot = [0] * 5
        if 1 <= label <= 5:
            one_hot[int(label) - 1] = 1
        return one_hot
    return [0, 0, 0, 0, 0]
```

### Hierarchical pooling

```python
class LobeAwarePooling(nn.Module):
    """Pool nodes per-lobe first, then aggregate across lobes."""
    
    def __init__(self, hidden_dim, num_lobes=5):
        super().__init__()
        self.lobe_attention = nn.Linear(hidden_dim, 1)
        self.lobe_transform = nn.Linear(hidden_dim, hidden_dim)
    
    def forward(self, x, batch, lobe_ids):
        # Step 1: Per-lobe mean pooling
        lobe_embs = []  # (B, 5, hidden)
        for lobe in range(5):
            mask = (lobe_ids == lobe)
            if mask.any():
                lobe_emb = global_mean_pool(x[mask], batch[mask])
            else:
                lobe_emb = torch.zeros(batch.max()+1, x.size(1))
            lobe_embs.append(lobe_emb)
        
        lobe_stack = torch.stack(lobe_embs, dim=1)  # (B, 5, H)
        
        # Step 2: Attention-weighted aggregation across lobes
        attn = torch.softmax(self.lobe_attention(lobe_stack), dim=1)
        global_emb = (attn * self.lobe_transform(lobe_stack)).sum(dim=1)
        
        return global_emb  # (B, H)
```
